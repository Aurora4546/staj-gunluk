## __init__() constructor içerisinde yazılan ve oluşturulan bütün değişkenler sınıfın değil o sınfın örneklerine ait olur.

# class metotları cls keyword ile sınıfın üyelerine erişebilir.
# __init__ metodu da self keyword ile instance üyelere erişir.
# Sınıf variable'larına instane üzerinden değiştirme yapılabilir.
class Node:
  cls_msg = "CLASS VAR"
  cls_mutable = [1,2,3,4,5]
  
  def __init__(self):
    self._msg = "instance var" 

  @property
  def msg(self):
    return self._msg
    
  @msg.setter
  def msg(self,value):
    self._msg = value
    
  @classmethod
  def func(cls):
    return cls.cls_msg

## Bütün sınıf variable'ları sadece sınıf referansları ile DEĞİŞTİRİLEBİLİR.
## Eğer sınıf üyelerinin değerini değiştirmeye çalışırsak instance için yeni attribute oluşturmuş oluruz.
## Class içindeki değeri değiştiremez.
## BU DURUM MUTABLE NESNELER İÇİN GEÇERLİ DEĞİLDİR. YANİ İNSTANCE İÇERİSİNDE YENİ BİR REFERANS EKLENİR ve
## BU REFERANS AYNI CLASS VARİABLE'IN TEMSİL ETTİĞİ NESNEYİ TEMSİL EDECEKTİR.
nde = Node()
nde.cls_msg = "test;" ## Yeni instance variable oluşturuldu.
print(nde.cls_msg)    ## instance variable kullanır
print(nde.func())     ## class variable kullanır
print(Node.cls_msg)   ## class variable
print(Node.func())
print(Node().func(),"\n")

nde.cls_mutable[0] = "TEST"
print(nde.cls_mutable)
print(Node.cls_mutable)

## class static ve instance metotlar class method OVERRIDE VAR
## Class method. Sınıfı ilgilendiren metotlardır. İnstance'lar da çağırabilir.
class MyClass:
    class_variable = 0

    def __init__(self, value):
        self.instance_variable = value

    # decorator oludğu için instance üzerinden de çağırılabilir.
    # avantajı cls keyword alması.
    @classmethod
    def class_method(cls, x):
        cls.class_variable += x
        return cls.class_variable

obj1 = MyClass(5)
obj2 = MyClass(10)

print(MyClass.class_method(3))  # Output: 3
print(MyClass.class_method(7))  # Output: 10

print(obj1.class_method(432))   # 10 + 432 class var değiştirildi.


## static metotlar OVERRIDE YOK
## sınıfa veya onun örneklerine herhangi bir şekilde bağlı olmayan metotlardır.
## Yani içerisinde self veya cls gibi keyword'leri alamazlar. Tamamen bağımsızlardır.
## Static metotlar instance'lar üzerinden çağırılabilirler ama herhangi bağımlılıkları yoktur.
## Class variable veya instance variable'lara erişemezler.

class MathOperations:
    class_var = "CLASS VAR"
    @staticmethod
    def add(x, y):
        return x + y
        
    @staticmethod
    def subtract(x, y):
        return x - y

print(MathOperations.add(5, 3))        
print(MathOperations.subtract(10, 4))

math = MathOperations()
print(math.add(2,5))


## İnstance metotları - OVERRIDE VAR
## Sınıfların örneklerine bağlı olan metotlardır. yani bunlar ile işlemler yaparlar.
## İnstance variable'larının tanımlanma yerleri __init__() metodu içerisidir.
## bu construcotr metot içerisinde instance variable'lar tanımlanabilirler.
## İnstance metotların içerisinde class variable'ları değiştirilemez VE OKUNAMAZ YANİ ERİŞİLEMEZ.

class Person:
    class_var = 10
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def introduce(self):
        return f"Hi, I'm {self.name} and I'm {self.age} years old."
        
    def change_class_var(self):
        print(class_var)
        
        
    

person1 = Person("Kishan", 20)

print(person1.introduce())
print(Person.introduce(person1))
print()


## İnheritance : python çoklu kalıtm destekler.
## AltSınıf(Kalıtım, kalıtım) şeklinde yapılabilir.

class A:
    def method(self):
        print("Method in class A")

class B(A):
    def method(self):
        super().method()  
        print("Method in class B")

class C(A):
    def method(self):
        super().method()  
        print("Method in class C")
        
## -------- super() constructor -----------------
## Base sınıfın constructor'ıdır
class Class():
    def __init__(self, x):
        print(x)

class SubClass(Class):
    def __init__(self, x):
        super().__init__(x)

x = [1, 2, 3, 4, 5]
a = SubClass(x)


## Multi inheritance desteklediği için sırası nasıl olacak.
## Soldan sağa sıralı gidecektir.
class Class1:
    def m(self):
        print("In Class1")

class Class2(Class1):
    def m(self):
        print("In Class2")
        super().m()

class Class3(Class1):
    def m(self):
        print("In Class3")
        super().m()

class Class4(Class2, Class3):
    def m(self):
        print("In Class4")   
        super().m()
     
print(Class4.mro())         
print(Class4.__mro__)
