shallow_copy = original_dict.copy()

# Dictionary with duplicate keys
# Dictionary'lerde anahatarların unique olması gerekir ama zorunlu değildir.
# eğer dupliacte anahtar varsa en son yani en sağdaki değer esas alınacaktır.
duplicated_keys = {"key1": "value1", "key1": "value2", "key1": "value3"}

print(duplicated_keys["key1"]) ## output ----> value3 en son bulunan değer.

harry_potter_dict = {    
    "Harry Potter": "Gryffindor",
    "Ron Weasley": "Gryffindor",
    "Hermione Granger": "Gryffindor"
}

add_characters_1 = {
    "Albus Dumbledore": "Gryffindor",
    "Luna Lovegood": "Ravenclaw"
}

add_characters_2 = [
    ["Draco Malfoy", "Slytherin"], 
    ["Cedric Diggory", "Hufflepuff"]
]

add_characters_3 = [
    ("Rubeus Hagrid", "Gryffindor"), 
    ("Minerva McGonagall", "Gryffindor")
]

# Merge dictionaries |= union operatörünü veya update metodunu kullanabiliriz.
harry_potter_dict |= add_characters_1
harry_potter_dict |= add_characters_2
harry_potter_dict |= add_characters_3
del harry_potter_dict["Minerva McGonagall"]


for x,y in harry_potter_dict.items():
  print(x,y)
  
# pop() ve popitem() metotları:
# pop() anahtarı verilmiş olan çifti siler ve geriye döndürür. Eğer yoksa default değer belirlenebilir.
# popitem() ise dictionary sonundaki değeri siler  ve döner.
## pop ve popitem metotolarının döndüğü değer tuple tipinde olacaktır. anahtar ve değer ikilileri tuple ile ifade edilebilir.
harry_potter_dict["Voldemort"] = "Slytherin"
silinen_tuple = harry_potter_dict.popitem()
silinen_tuple_2 = harry_potter_dict.pop("OLMAYAN KEY", "Default deger" )
print(silinen_tuple, silinen_tuple_2)

## Dictionary'lerde bracetlerle erişmeye çalıştığımız zaman anahtar yoksa error verir.
## Bunu önlemek için get() metodu kullanlabilir. Eğer anahtar yoksa default değer döner.
print(harry_potter_dict.get("VOLDEMORT", "default deger"))

## Eğer anahtarın bulunmazsa ve yeni değer eklemek istersek setdefault() kullanılabilir.
## anahtarın dictionary içerisinde olup olmadığından emin olamadığımız zaman mantıklı
harry_potter_dict.setdefault("BULUNAMADI" , "VALUE")
print(harry_potter_dict["BULUNAMADI"])

## items() keys() values() metotlarının hepsi yeni değer dönerler.
## dict_items() nesnesi tuple'lardan oluşan bir listedir.
print(f"keys() --> {harry_potter_dict.keys()}\nvalues() --> {harry_potter_dict.values()}\nitems() --> {harry_potter_dict.items()}")

## Dictionary comprehension
## listelerde ve tuple'larda olduğu gibi kısa yoldan döngü ile Dictionary oluşturulabilir.
## dictionary'ler listelere göre daha performanslı çalışırlar. 
## comprehension : comprehension in Python means applying the same operation on each element of an iterable
## comprehension kullandığımız zaman yeni değer döner.
dict_comp = {x:v.strip().upper() for x,v in harry_potter_dict.items()}
print(dict_comp)
