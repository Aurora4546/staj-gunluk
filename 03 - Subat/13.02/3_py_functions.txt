## Arbitrary parametreler
# ne kadar deđer gönderileceđi belirli olmayan metotlarda kullanýlabilir.
# örnek olarak print() metoduna kaç tane veri geleceđi belirli deđil.
# Bu parametreler artýk liste içeriisnde tutulacaktýr. Fonksiyon içerisinde buna uygun iţlemlerin yapýlmasý gerekir.
def myFun(*args, **kwargs):
    print("Non-Keyword Arguments (*args):")
    for arg in args:
        print(arg)

    print("\nKeyword Arguments (**kwargs):")
    for key, value in kwargs.items():
        print(f"{key} == {value}")

# Function call with both types of arguments
myFun('Hey', 'Welcome', first='Geeks', mid='for', last='Geeks')

## Global nonlocal 
# nonlocal keyword INNER FUNCTION'lar için kullanýlýr. Ýnner içerisindeki tanýmlanan ve ismi dýţ function içerisindeki
# tanýmlanan baţka bir deđiţken ile ayný olan deđiţkenlerin dýţ fonksiyon içerisindekileri temsil etmesi için kullanýlýr.
# Global keyword ise inner function durumlarý olmadýđý zaman kullanýlýr.
def outer():
    x = "local"
    
    def inner():
        nonlocal x
        x = "nonlocal"
        print("inner:", x)
    
    inner()
    print("outer:", x)
outer()

    def wrapper(*args, **kwargs):
        print("Before execution")
        result = func(*args, **kwargs)
        print("After execution")
        return result
    return wrapper

def add(a, b):
    return a + b

## Method decorator.
## nesnelerin metotlarýný deđiţtirmek için kullanýlýr.
def method_decorator(func):
    def wrapper(self, *args, **kwargs):
        print("Before method execution")
        res = func(self, *args, **kwargs)
        print("After method execution")
        return res
    return wrapper

class MyClass:
    def say_hello(self):
        print("Hello!")
    say_hello = method_decorator(say_hello)
obj = MyClass()
obj.say_hello()

## Property metotlarý normal field'mýţ gibi ERÝŢÝLMESÝNE izin verir.
# ayný ţekilde atamalarda da normal field'mýţ gibi olmasý için @radius.setter
# annotation kullanýlabilir.
class Circle:
    def __init__(self, radius):
        self._radius = radius

    @property
    def radius(self):
        return self._radius

    @radius.setter
    def radius(self, value):
        if value >= 0:
            self._radius = value
        else:
            raise ValueError("Radius cannot be negative")

    @property
    def area(self):
        return 3.14159 * (self._radius ** 2)

# Using the property
c = Circle(5)
print(c.radius) 
print(c.area)    
c.radius = 10
print(c.area)

## Python closures
## Closures'dan kasýt fonksiyon içerisindeki diđer fonksiyondur.
## closure olma durumlarý : Dýţ fonksiyonun içerideki fonksiyonu deđer olarak döndürmesi
## Ýç fonksiyonun, dýţarýdaki fonksiyonun deđerini kullanmasý

def outer_function(x):
    # Outer function: takes 'x' and defines inner_function
    def inner_function(y):
        return x + y  # 'x' is remembered from outer_function
    return inner_function  # Returns inner function (closure)

# Create a closure with x = 10
closure = outer_function(10)

# Call the closure with different values of 'y'
print(closure(5)) 
print(closure(20))


## Dýţ metodundaki deđeri counter deđiţken olarak kullanan bir iç fonksiyon.
## Ýçerideki fonksiyon dýţ fonksiyondaki deđeri kulanýyor ve deđerini hatýrlýyor.
# Bunu yapabilmek iiçn nonlocal keyword kullanýlmasý gerekiyor.
def make_counter():
    count = 0  # This variable will be remembered
    def counter():
        nonlocal count  # Modify outer variable
        count += 1
        return count
    return counter

counter1 = make_counter()
print(counter1())  # 1
print(counter1())

## Closure örneđi
def pre(p):
    # Outer function stores prefix
    def add(t):
        # Inner function uses stored prefix
        return p + " " + t
    return add  # Return closure

# Create a closure that always prefixes with "Hello"
h = pre("Hello")
print(h("World"))
print(h("Python"))
