## En son üstüne inţa edebileceđi bir taným lazým.
def fact(value):
  if value == 0 :
    return 1
  else :
    return value * fact(value - 1)
  
print(fact(5))

def fibonacci(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(10))
